# Define as variáveis
$assunto = "Relatorio Risco"
$destinatario = "risco@granitopagamentos.com.br"
$corpo = "Segue relatório em anexo."

# Envia o e-mail utilizando o método Send-MailMessage
Send-MailMessage -From 'portal@granitopagamentos.com.br' -To $destinatario -Subject $assunto -Body $corpo

# Verifica se o e-mail foi enviado com sucesso e retorna o código de saída
if ($LASTEXITCODE -eq 0) {
    Write-Host "E-mail enviado com sucesso."
} else {
    Write-Host "Erro ao enviar o e-mail."
}
